<input_description> ::= <db_name> ';' <container_name> ';' <container_class> ';' <template_arg> [ ';' <template_arg> ]

<db_name> ::= 'whatever.db'

<container_name> ::= 'any_name'

<container_class> ::= 'list' | 'vector' | 'deque' | 'set' | 'multiset' | 'hash_set' | 'hash_multiset' | 'map' | 'multimap' | 'hash_map' | 'hash_multimap'

<template_arg> ::=  <class_definition> { ':' <class_definition> }

<class_definition> ::= <attribute_name> ',' <primitive> | 
		   <class_name> ',' 'class' '-' <attribute_name> ',' <type> { '-' <attribute_name> ',' <type> }

	     	 		     <!-- for clarity -->
